// Comando para creación de archivo 
// de configuración del proyecto package.json
npm init -y

// Instalar dependencia de TYPESCRIPT
npm install typescript --save-dev

// Generar archivo de configuración de Typescript
npx tsc --init

// Comando de compilación de todos los Typescript
// contenidos en el directorio
npx tsc

// Ejecución del programa
node index.js
